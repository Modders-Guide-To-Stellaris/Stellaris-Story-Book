building_coven_castle_1 = {
    base_buildtime = @b1_time
    can_build = no
    category = government
    can_be_disabled = no
    upgrades = { "building_covenant_2" }

	potential = {
		exists = owner
		owner = {
			is_country_type = default
		}
		NOR = {
			is_planet_class = pc_hive
			is_planet_class = pc_machine
		}
		is_artificial = no
	}

	allow = {
		has_upgraded_capital = yes
	}

	destroy_trigger = {
		OR = {
			is_planet_class = pc_hive
			is_planet_class = pc_machine
	}

	resources = {
		category = planet_buildings
		cost = {
			energy = 1500
			multiplier = value:gaiaseeder_cost_mult
		}
		upkeep = {
            unity = 5
			multiplier = value:gaiaseeder_upkeep
		}
	}

	    planet_modifier = {
		    job_coven_head_add = 1
	    }
    }
}
