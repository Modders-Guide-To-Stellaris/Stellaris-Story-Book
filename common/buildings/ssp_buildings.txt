building_coven_castle_1 = {
	base_buildtime = @b1_time
	can_build = no
	category = government
	can_be_disabled = no
	upgrades = { "building_coven_castle_2" }

	potential = {
		exists = owner
		owner = {
			AND = {
				has_valid_civic = civic_ssp_coven_system
				is_country_type = default
			}
		}
		NOR = {
			is_planet_class = pc_hive
			is_planet_class = pc_machine
		}
		is_artificial = no
	}

	allow = {
		has_upgraded_capital = yes
	}

	destroy_trigger = {
		OR = {
			is_planet_class = pc_hive
			is_planet_class = pc_machine
		}
	}
	on_built = {
		random_list = {
			16 = {
				set_planet_flag = ssp_job.1#healing
			}
			17 = {
				set_planet_flag = ssp_job.2#plant
			}
			17 = {
				set_planet_flag = ssp_job.4#construction
			}
			16 = {
				set_planet_flag = ssp_job.5#illusion
			}
			17 = {
				set_planet_flag = ssp_job.6#oracle
			}
			17 = {
				set_planet_flag = ssp_job.7#bard
				}
			}
		}
	}
