			is_gestalt = yes

		potential = {

		}

		army_defense_damage_mult = 0.25
		army_defense_health_mult = 0.25
	}


	
	
	score = 1000

	active_effect = {
		add_modifier = {
			modifier = "ssp_r_animarum_active_effect"
			days = 3600
		}

	
		custom_tooltip = "ssp_r_animarum_active_effect_tooltip"
		hidden_effect = {
			capital_scope = {
				create_fleet = {
					name = "NAME_reanimated_army_fleet"
					effect = {
						set_owner = root
						while = { 
						count = 5
						create_army_transport = {
							ship_name = "The Undead Army"
							army_name = "Seeds of the Undead"
							army_type = ssp_reanimated_army
						}
						}
						set_location = {
							target = prev
							distance = 5
							angle = random 
						}
					}
					settings = {
						can_change_leader = yes
						uses_naval_capacity = no
						spawn_debris = no
					}
				}
			}
		}
		custom_tooltip = relic_triumph_cooldown
		hidden_effect = {
			add_modifier = {
				modifier = "relic_activation_cooldown" 
				days = 1800
			}		
		}
	

	# Possible check for activation
	possible = {
		custom_tooltip = {
			fail_text = "requires_relic_no_cooldown"
			NOT = { 
				has_modifier = relic_activation_cooldown 
			}
		}
	}
  }
