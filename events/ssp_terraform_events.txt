namespace = ssp_terraform

planet_event = {
    id = ssp_terraform.1
    is_triggered_only = yes
    hide_window = yes

    immediate = {
        switch = {
            trigger = is_planet_class
            pc_desert = { add_modifier = { modifier = ssp_ecology_desert}}
            pc_arid = { add_modifier = { modifier = ssp_ecology_arid }}
            pc_savannah = { add_modifier = { modifier = ssp_ecology_savannah}}
            pc_ocean = { add_modifier = { modifier = ssp_ecology_ocean}}
            pc_continental = { add_modifier = { modifier = ssp_ecology_continental}}
            pc_tropical = { add_modifier = { modifier = ssp_ecology_tropical}}
            pc_arctic = { add_modifier = { modifier = ssp_ecology_arctic}}
            pc_alpine = { add_modifier = { modifier = ssp_ecology_alpine}}
            pc_tundra = { add_modifier = { modifier = ssp_ecology_tundra}}
            pc_gaia = { add_modifier = { modifier = ssp_ecology_gaia}}
        }
        change_pc = pc_ssp_ecoworld
    }
}

planet_event = {
    id = ssp_terraform.2
    is_triggered_only = yes
    hide_window = yes

    immediate = {
        if = {
            limit = { is_planet_class = pc_ssp_ecoworld }
            if = {
                limit = {
                    AND = {
                        has_building = ssp_ecology_unification_centre
                        has_modifier = ssp_ecology_unify_centre_destroyed
                    }
                }
                remove_modifier = ssp_ecology_unify_centre_destroyed
            }
            if = {
                limit = {
                    AND = {
                        has_planet_flag = EcologyCentreBuilt
                        NOT = { has_building = ssp_ecology_unification_centre }
                    }
                }
                remove_planet_flag = EcologyCentreBuilt
                add_modifier = { modifier = ssp_ecology_unify_centre_destroyed days = -1}
            }
        }
    }
}
