namespace = ssp_planet

planet_event = {
    id = ssp_planet.100
    is_triggered_only = yes
    hide_window = yes

    trigger = {
        has_orbital_bombardment = yes
    }

    immediate = {
        every_fleet_in_orbit = {
            limit = {
                owner = { #owner of fleet is at war with the owner of the planet
                    is_at_war_with = root.owner
                }
            }
            set_fleet_flag = ssp_under_fire_from_planet
        }

        set_variable = {
            which = ssp_planetary_anti_air
            value = 0
        }

        change_variable = {
            which = ssp_planetary_anti_air
            value = value:ssp_planetary_antiair_damage
        }
    }
}
